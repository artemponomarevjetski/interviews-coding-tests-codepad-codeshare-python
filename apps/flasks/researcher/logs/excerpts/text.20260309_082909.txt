def _parse_tags_csv(tags_csv: str) -> List[str]:
    raw = re.split(r"[\n,]+", tags_csv or "")
    return [x.strip() for x in raw if x and x.strip()]


def _parse_json_dict(raw: str) -> Dict[str, str]:
    raw = (raw or "").strip()
    if not raw:
        return {}
    obj = json.loads(raw)
    if not isinstance(obj, dict):
        raise ValueError("classifications_json must decode to a JSON object/dict.")
    return {str(k): str(v) for k, v in obj.items()}

def _try_call_patterns(sync_fn, domain: str, catalog: str, schema: str, obj: str,
                       tags: List[str], classifications: Dict[str, Any], is_starburst: bool):
    """
    Try multiple call signatures to tolerate minor API differences:
    - isStarburst=...
    - is_starburst=...
    - with or without tags/classifications
    """
    patterns = [
        ("domain,catalog,schema,obj,tags,cls,isStarburst=..",
         lambda: sync_fn(domain, catalog, schema, obj, tags, classifications, isStarburst=is_starburst)),
        ("domain,catalog,schema,obj,tags,cls,is_starburst=..",
         lambda: sync_fn(domain, catalog, schema, obj, tags, classifications, is_starburst=is_starburst)),
        ("domain,catalog,schema,obj,tags,cls",
         lambda: sync_fn(domain, catalog, schema, obj, tags, classifications)),
        ("domain,catalog,schema,obj,isStarburst=..",
         lambda: sync_fn(domain, catalog, schema, obj, isStarburst=is_starburst)),
        ("domain,catalog,schema,obj,is_starburst=..",
         lambda: sync_fn(domain, catalog, schema, obj, is_starburst=is_starburst)),
        ("domain,catalog,schema,obj",
         lambda: sync_fn(domain, catalog, schema, obj)),
    ]

    last_err = None
    for label, fn in patterns:
        try:
            result = fn()
            print(f"✅ sync_platform.sync succeeded using pattern: {label}")
            return result
        except TypeError as e:
            last_err = e
        except Exception:
            raise

    raise TypeError(f"Unable to call sync_platform.sync with any known signature. Last TypeError: {last_err}")

def _sql_literal(s: str) -> str:
    return "'" + s.replace("'", "''") + "'"


def _quote_ident(part: str) -> str:
    part = part.replace("`", "``")
    return f"`{part}`"


def _normalize_fqn(catalog: str, schema: str, object_name: str) -> Tuple[str, str, str]:
    """
    Accepts:
      - object
      - schema.object
      - catalog.schema.object
    """
    parts = [p for p in object_name.split(".") if p]
    if len(parts) == 1:
        if not catalog or not schema:
            raise ValueError("If object_name is not fully qualified, catalog and schema must be provided.")
        return catalog, schema, parts[0]
    if len(parts) == 2:
        if not catalog:
            raise ValueError("If object_name is schema.object, catalog must be provided.")
        return catalog, parts[0], parts[1]
    if len(parts) == 3:
        return parts[0], parts[1], parts[2]
    raise ValueError("object_name must be in form object, schema.object, or catalog.schema.object")


def _detect_object_type(catalog: str, schema: str, obj: str) -> str:
    df = spark.sql(f"""
        SELECT table_type
        FROM system.information_schema.tables
        WHERE table_catalog = {_sql_literal(catalog)}
          AND table_schema  = {_sql_literal(schema)}
          AND table_name    = {_sql_literal(obj)}
        LIMIT 1
    """)
    rows = df.collect()
    if not rows:
        raise ValueError(f"Object not found in system.information_schema.tables: {catalog}.{schema}.{obj}")
    table_type = (rows[0]["table_type"] or "").upper()
    if "VIEW" in table_type:
        return "VIEW"
    return "TABLE"


def _get_current_owner(catalog: str, schema: str, obj: str):
    df = spark.sql(f"""
        SELECT table_owner
        FROM system.information_schema.tables
        WHERE table_catalog = {_sql_literal(catalog)}
          AND table_schema  = {_sql_literal(schema)}
          AND table_name    = {_sql_literal(obj)}
        LIMIT 1
    """)
    rows = df.collect()
    if not rows:
        return None
    return rows[0]["table_owner"]

ownership_payload: Dict[str, Any] = {
    "status": "SKIPPED",
    "reason": "transfer_ownership=false"
}

if not sync_ok:
    ownership_payload = {
        "status": "SKIPPED",
        "reason": "sync failed"
    }

elif transfer_ownership:
    if not new_owner:
        ownership_payload = {
            "status": "FAILED",
            "reason": "transfer_ownership=true but new_owner is empty"
        }
    else:
        try:
            catalog_n, schema_n, obj_n = _normalize_fqn(catalog, schema, object_name)
            fqn_sql = f"{_quote_ident(catalog_n)}.{_quote_ident(schema_n)}.{_quote_ident(obj_n)}"

            final_type = object_type
            if final_type == "AUTO":
                final_type = _detect_object_type(catalog_n, schema_n, obj_n)

            if final_type not in {"TABLE", "VIEW"}:
                raise ValueError(f"Unsupported object_type: {final_type}. Expected AUTO/TABLE/VIEW.")

            owner_sql = _quote_ident(new_owner)
            if final_type == "VIEW":
                stmt = f"ALTER VIEW {fqn_sql} OWNER TO {owner_sql}"
            else:
                stmt = f"ALTER TABLE {fqn_sql} OWNER TO {owner_sql}"

            current_owner_before = _get_current_owner(catalog_n, schema_n, obj_n)

            print("=== Ownership transfer parameters ===")
            print(json.dumps({
                "catalog": catalog_n,
                "schema": schema_n,
                "object": obj_n,
                "resolved_fqn_sql": fqn_sql,
                "object_type": final_type,
                "new_owner": new_owner,
                "dry_run": dry_run,
                "verify": verify,
                "current_owner_before": current_owner_before,
                "sql": stmt,
            }, indent=2))

            if dry_run:
                print("DRY RUN: not executing ownership change.")
                ownership_payload = {
                    "status": "DRY_RUN",
                    "catalog": catalog_n,
                    "schema": schema_n,
                    "object_name": obj_n,
                    "object_type": final_type,
                    "new_owner": new_owner,
                    "current_owner_before": current_owner_before,
                    "sql": stmt,
                }
            else:
                spark.sql(stmt)
                print("✅ Ownership transfer executed.")

                current_owner_after = _get_current_owner(catalog_n, schema_n, obj_n)
                ownership_payload = {
                    "status": "SUCCESS",
                    "catalog": catalog_n,
                    "schema": schema_n,
                    "object_name": obj_n,
                    "object_type": final_type,
                    "new_owner": new_owner,
                    "current_owner_before": current_owner_before,
                    "current_owner_after": current_owner_after,
                    "sql": stmt,
                }

                if current_owner_after != new_owner:
                    ownership_payload["status"] = "FAILED"
                    ownership_payload["reason"] = (
                        f"Ownership verification failed. Expected '{new_owner}', got '{current_owner_after}'."
                    )

                if verify:
                    print("=== DESCRIBE EXTENDED (verification) ===")
                    display(spark.sql(f"DESCRIBE EXTENDED {fqn_sql}"))
                else:
                    print("verify=false; skipping DESCRIBE EXTENDED")

        except Exception as e:
            ownership_payload = {
                "status": "FAILED",
                "error_type": type(e).__name__,
                "error_message": str(e),
                "traceback": traceback.format_exc()[:15000],
            }
            print("❌ Ownership transfer failed")
            print(json.dumps(ownership_payload, indent=2))
