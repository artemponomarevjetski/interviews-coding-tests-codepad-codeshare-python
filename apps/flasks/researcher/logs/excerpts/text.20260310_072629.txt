class ServerError(Exception):
    """Raised on 5xx errors."""
    def __init__(self, status_code: int, message: str):
        self.status_code = status_code
        self.message = message
        super().__init__(f"{status_code}: {message}")

class RateLimitError(Exception):
    """Raised on 429 errors."""
    def __init__(self, retry_after: int = None):
        self.retry_after = retry_after
        super().__init__(f"Rate limited. Retry after: {retry_after}s")

def retry_on_server_error():
    return retry(
        # Stop conditions - don't retry forever
        stop=stop_after_attempt(4), # 4 attempts total (1 initial + 3 retries)

        # Wait strategy - exponential with jitter to avoid thundering herd
        wait=wait_exponential_jitter(
            initial=3, # First retry after ~3 seconds
            max=45, # Cap at 45 seconds between retries
            jitter=5 # Add random 0-5 seconds to spread out retries
        ),

        # What to retry - only transient errors
        retry=retry_if_exception_type((
            requests.exceptions.Timeout,
            requests.exceptions.ConnectionError,
            ServerError,
            RateLimitError,
        )),

        # Logging
        before_sleep=before_sleep_log(logger, logging.WARNING),
        after=after_log(logger, logging.DEBUG),

        # Re-raise the original exception after all retries exhausted
        reraise=True
        )

def _request(self, method: str, endpoint: str, timeout: int = 120, **kwargs) -> dict:
    """Make request with proper error categorization."""
    response = self.session.request(
        method,
        self._api_url(endpoint),
        headers=self.headers,
        timeout=timeout,
        **kwargs
    )

    # Rate limit - might want special handling
    if response.status_code == 429:
        retry_after = response.headers.get("Retry-After")
        raise RateLimitError(int(retry_after) if retry_after else None)

    # Server errors - transient, worth retrying
    if response.status_code >= 500:
        raise ServerError(response.status_code, response.text[:200])

    # Client errors (4xx except 429) - don't retry, our fault
    response.raise_for_status()

    return response.json() if response.text else {}

class HttpClient(ABC):
    def __init__(self, config: BaseModel):
        self.config = config
        self.session = requests.Session()

        # Set timeout
        self.timeout = config.timeout_seconds

        # Set headers (to be implemented by subclasses)
        self.session.headers.update(self._get_headers())

        # Set proxy if applicable
        if hasattr(config, "proxy_url") and config.proxy_url:
            self.session.proxies = {
                "http": config.proxy_url,
                "https": config.proxy_url,
            }

        # Set CA certificate for SSL verification
        if hasattr(config, "ca_cert") and config.ca_cert:
            self.session.verify = config.ca_cert
        else:
            self.session.verify = True  # Default to verifying SSL certificates

    @abstractmethod
    def _get_headers(self) -> Dict[str, str]:
        """
        Abstract method to define headers for the API client.
        Must be implemented by subclasses.
        """
        pass

    def should_retry(response):
        """
        Retry on server errors:
        408 Request Timeout: The server timed out waiting for your request
        409 Conflict: Usually means the request conflicts with the current state of a resource (e.g., an edit collision). Only retry if you have logic to resolve the state difference
        429 Too Many Requests: You are being rate-limited. Wait for the duration specified in the Retry-After header before trying again to avoid a harder block
        500 Internal Server Error: This is a "catch-all" error. It might be a temporary glitch (worth a retry) or a permanent bug in the request. Use exponential backoff if you choose to retry this.
        502 Bad Gateway: A communication error between backend servers
        503 Service Unavailable: The server is temporarily overloaded or down for maintenance.
        504 Gateway Timeout: An upstream server failed to respond in time.
        
        Never Retry (Account Security Risk):
        401 Unauthorized: Your credentials (password, API key, token) are wrong or missing. Do not retry without manually fixing the credentials
        403 Forbidden: The server understands who you are but refuses to let you in. Retrying will not change this and may be flagged as an attack
        423 Locked: Specifically used (mostly in WebDAV) to indicate a resource is locked. If your account is already locked, retrying will not unlock it.
        """
        return response.status_code in [408, 409, 429, 500, 502, 503, 504]
    
    # Retry somethimes will lock the account
    # @retry(
    #     stop=stop_after_attempt(4),  # 4 attempts total (1 inital + 3 retries)
    #     wait=wait_exponential(
    #         multiplier=1, 
    #         min=2, 
    #         max=10
    #         ),
    #     retry=retry_if_exception_type(should_retry)
    # )
    def request(self, method: str, endpoint: str, **kwargs) -> Optional[Dict[str, Any]]:
        """
        Synchronous method to make HTTP requests with retry logic.
        """
        url = f"{self.config.host}/{endpoint.lstrip('/')}"
        try:
            response = self.session.request(method, url, timeout=self.timeout, **kwargs)
            response.raise_for_status()
            return response.json()
        except RequestException as e:
            logging.error(f"HTTP request failed: {e}")
            raise  # Raise the exception to trigger retry

    def get(
        self, endpoint: str, params: Optional[Dict[str, Any]] = None
    ) -> Optional[Dict[str, Any]]:
        return self.request("GET", endpoint, params=params)

    def post(
        self,
        endpoint: str,
        json_data: Optional[Dict[str, Any]] = None,
        params: Optional[Dict[str, Any]] = None,
    ) -> Optional[Dict[str, Any]]:
        # If a custom JSON encoder is needed, serialize manually and use the `data` parameter
        if json_data:
            json_data = json.dumps(json_data, cls=CustomJSONEncoder)
            return self.request("POST", endpoint, data=json_data, params=params)
        # Otherwise, let `requests` handle the serialization
        return self.request("POST", endpoint, json=json_data, params=params)

    def put(
        self, endpoint: str, json_data: Optional[Dict[str, Any]] = None
    ) -> Optional[Dict[str, Any]]:
        # If a custom JSON encoder is needed, serialize manually and use the `data` parameter
        if json_data:
            json_data = json.dumps(json_data, cls=CustomJSONEncoder)
            return self.request("PUT", endpoint, data=json_data)
        # Otherwise, let `requests` handle the serialization
        return self.request("PUT", endpoint, json=json_data)
    
    def patch(
        self, endpoint: str, json_data: Optional[Dict[str, Any]] = None,
        params: Optional[Dict[str, Any]] = None,
    ) -> Optional[Dict[str, Any]]:
        if json_data:
            json_data = json.dumps(json_data, cls=CustomJSONEncoder)
        return self.request("PATCH", endpoint, json=json_data, params=params)

    def delete(
        self, endpoint: str, params: Optional[Dict[str, Any]] = None
    ) -> Optional[Dict[str, Any]]:
        return self.request("DELETE", endpoint, params=params)

def is_valid_uuid(uuid_string):
    try:
        uuid_obj = uuid.UUID(uuid_string, version=4)

class AsyncManager:
    """Orchestrates concurrent execution of Immuta and Collibra sync tasks."""

    def __init__(self):
        self.immuta_task = ImmutaTask()
        self.collibra_task = CollibraTask()

    async def run(self) -> list[TaskResult]:
        """Run both tasks concurrently and return all results."""
        logger.info("AsyncManager starting concurrent sync...")

        results = await asyncio.gather(
            self.immuta_task.run(),
            self.collibra_task.run(),
            return_exceptions=False,
            )

        self._log_summary(results)
        return results

    async def run_with_timeout(self, timeout: float = 30.0) -> list[TaskResult]:
        """Run both tasks with a global timeout."""
        try:
            return await asyncio.wait_for(self.run(), timeout=timeout)
        except asyncio.TimeoutError:
            logger.error(f"AsyncManager timed out after {timeout}s")
            return [
                TaskResult("ImmutaTask", TaskStatus.FAILED, error=TimeoutError()),
                TaskResult("CollibraTask", TaskStatus.FAILED, error=TimeoutError()),
            ]

    def _log_summary(self, results: list[TaskResult]):
        for r in results:
            if r.status == TaskStatus.SUCCESS:
                logger.info(f"{r.task_name}: {r.details}")
            else:
                # Use details to decide retry strategy
                logger.error(f"{r.task_name} failed at: {r.details.get('failed_at')}")

        failed = [r for r in results if r.status == TaskStatus.FAILED]
        if failed:
            names = ", ".join(r.task_name for r in failed)
            logger.warning(f"Sync completed with failures: {names}")
        else:
            logger.info("All sync tasks completed successful")

def get_last_element(path: str) -> str:
    """
    Retrieve the last element from a '>'-delimited string.

    Args:
        path (str): The input string in the format 'element1>element2>element3>element4'.

    Returns:
        str: The last element in the string.
    """
    if not path or '>' not in path:
        raise ValueError("Input must be a non-empty string containing '>' as a delimiter.")
    
    # Split the string by '>' and return the last element
    return path.split('>')[-1]

def create_delta_table(
    catalog: str,
    schema: str,
    table: str,
    table_columns: str,
    comment: str,
    tblproperties: Optional[Dict[str, str]] = None,
):
    """
    Creates a Delta table with the specified catalog, schema, table name, schema definition, comment, and table properties.

    Args:
        catalog (str): The catalog name.
        schema (str): The schema name.
        table (str): The table name.
        table_schema (str): The schema definition for the table (e.g., column definitions).
        comment (str): The comment to add to the table.
        tblproperties (Optional[Dict[str, str]]): A dictionary of table properties (key-value pairs).

    Returns:
        None
    """
    # Convert table properties dictionary to a string format for SQL
    tblproperties_str = ", ".join(
        [f"'{key}' = '{value}'" for key, value in tblproperties.items()]
    ) if tblproperties else ""

    # Construct the SQL statement
    create_table_sql = f"""
        CREATE TABLE IF NOT EXISTS `{catalog}`.`{schema}`.`{table}` (
            {table_columns}
        ) USING DELTA
        COMMENT '{comment}'
        TBLPROPERTIES ({tblproperties_str})
    """

    # Execute the SQL statement
    try:
        spark.sql(create_table_sql)
        logger.info(f"Table `{catalog}`.`{schema}`.`{table}` created successfully.")
    except Exception as e:
        logger.error(f"Error creating table `{catalog}`.`{schema}`.`{table}`: {e}")

def get_classifications_and_tags(catalog: str, schema: str, table: str):
    """
    Retrieves classifications and tags from the comment of a specified table.

    Args:
        catalog (str): The catalog name.
        schema (str): The schema name.
        table (str): The table name.

    Returns:
        Tuple[Dict[str, str], List[str]]: A tuple containing classifications (key-value pairs) and tags (list).
    """

    classifications = {}
    tags = []
    
    try:
        # Retrieve the comment from the table
        comment_df = spark.sql(
            f"""
            SELECT comment
            FROM system.information_schema.tables
            WHERE table_catalog = '{catalog}'
            AND table_schema = '{schema}'
            AND table_name = '{table}'
            """
        )
    except AnalysisException as e:
        logger.error(f"Error retrieving comment: {e}")
        return classifications, tags

    # Check if the comment exists
    if comment_df is None or comment_df.count() == 0:
        logger.warning(f"No comment found for table {catalog}.{schema}.{table}")
        return classifications, tags

    # Retrieve the first row safely
    first_row = comment_df.limit(1).collect()[0]
    comment = first_row["comment"] if "comment" in first_row else None

    if not comment:
        logger.warning(f"No comment found for table {catalog}.{schema}.{table}")
        return classifications, tags

    logger.info(f"Retrieved comment: {comment}")

    # Parse the key-value pairs and tags
    def parse_comment(comment_str):
        # Splits by comma only when they are outside of square brackets
        pattern = r",(?![^\[]*\])"
        pairs = re.split(pattern, comment_str)
        for pair in pairs:
            pair = pair.strip()  # Remove leading/trailing whitespace
            if ":" in pair:
                key, value = pair.split(":", 1)
                classifications[key.strip()] = value.strip()
            else:
                tags.append(pair)
        return classifications, tags

    try:
        classifications, tags = parse_comment(comment)
        # logger.info(f"Classifications: {classifications}")
        # logger.info(f"Tags: {tags}")
    except Exception as e:
        logger.error(f"Error parsing comment: {e}")

    return classifications, tags

class CollibraException(Exception):
    pass

class CollibraClient(HttpClient):
    def __init__(self, config: CollibraConfig):
        """
        Initialize the CollibraClient with the given configuration.

        Args:
            config (CollibraConfig): Configuration for Collibra API.
        """
        super().__init__(config)  # Call the parent class's constructor

    def _get_headers(self) -> Dict[str, str]:
        """
        Generate headers for Collibra API requests.

        Returns:
            Dict[str, str]: Headers including basic auth and content type.
        """
        auth_str = f"{self.config.username}:{self.config.password}"
        encoded_auth = base64.b64encode(auth_str.encode("utf-8")).decode("utf-8")
        return {
            "Authorization": f"Basic {encoded_auth}",
            "Content-Type": "application/json",
            "Accept": "application/json",
        }

    def get_domain_id(self, domain: str) -> Optional[UUID]:
        """
        Retrieve the domain ID for a given domain name.

        Args:
            domain (str): The name of the domain.

        Returns:
            Optional[UUID]: The UUID of the domain if found, otherwise None.
        """
        params = {
            "offset": 0,
            "limit": 1,
            "countLimit": -1,
            "name": domain,
            "nameMatchMode": "EXACT",
            "excludeMeta": "true",
            "includeSubCommunities": "false"
            }
        
        endpoint = "rest/2.0/domains"
        
        try:
            logger.info(f"Retrieving domain id for domain {domain}")
            data = self.get(endpoint, params=params)

            if data:
                domain_id = uuid.UUID(data["results"][0].get("id"))
                return domain_id
            else:
                logger.warning(f"No domain found with name: {domain}")
                return None
        except CollibraException as e:
            logger.error(f"Failed to retrieve domain id for domain {domain}': {e}")
            return None
        except Exception as e:
            logger.error(f"Unexpected error while retrieving domain id for domain {domain}: {e}")
            return None
    
    def get_asset_by_full_path(self, full_path:str, domain_id: UUID) -> Optional[Asset]:
        params = {
            "name": full_path,
            "nameMatchMode": "END",
            "domainId": domain_id,
            "limit": 1,
            "typeInheritance": "true",
            "excludeMeta": "true",
            "sortField": "NAME"
        }

        try:
            data = self.get("rest/2.0/assets", params=params)
            logger.info(f"{full_path} Asset results: {json.dumps(data, indent=4)}")
            results = data.get("results", [])
            res = results[0]
            return Asset(
                id=res["id"],
                full_name=res["name"],
                display_name=res["displayName"],
                domain_id=res["domain"]["id"],
                domain_name=res["domain"]["name"],
                type_id=res["type"]["id"],
                type_name=res["type"]["name"],
                status_id=res["status"]["id"],
                status_name=res["status"]["name"],
            )
        except CollibraException as e:
            logger.error(f"Asset lookup failed: {e}")
            return None

    def list_tags(self, name:str = None) -> Dict[str, Dict]:
        """
        List all tags in Collibra.
        
        GET /tags
        
        Query Parameters:
        - offset: Pagination offset
        - limit: Max results per page
        - name: Filter by tag name
        - nameMatchMode: EXACT, START, END, ANYWHERE (default: ANYWHERE)
        
        Response:
        {
            "total": 123,
            "offset": 0,
            "limit": 100,
            "results": [
                {"id": "uuid", "name": "tag_name", "createdBy": "uuid", ...}
            ]
        }
        
        Returns:
            Dict mapping tag names (lowercase) to tag details
        """
        try:
            all_tags = [] # To store all retrieved tags
            offset = 0 # Pagination offset
            limit = 100 # Number of results per page
            countLimit = -1 # Allows to limit the number of elements that will be counted. -1 will count everything and 0 will cause the count to be skipped
            
            while True:
                params = {"offset": offset, "limit": limit, "countLimit": countLimit}
                if name:
                    params["name"] = name
                    params["nameMatchMode"] = "ANYWHERE"
                
                response = self.get("rest/2.0/tags", params=params)
                
                # logger.info(f"collibra tags results: {response}")
                data = response
                results = data.get('results', [])
                all_tags.extend(results)
                
                # Check if there are more pages
                total = data.get('total', len(results))
                if offset + limit >= total:
                    break
                offset += limit
            
            result = {tag['name'].lower(): tag for tag in all_tags}

            logger.info(f"Retrieved {len(result)} tags from Collibra")
            return result
        except CollibraException as e:
            logger.error(f"Failed to list Collibra tags: {e}")
            return None

    def tag_exists(self, tag_name: str) -> bool:
            """Check if a tag exists in Collibra."""
            tags = self.list_tags()
            return tag_name.lower() in tags

    def create_tag(self, asset_id: str, tag: str) -> Dict:
        """
        Create new tag in Collibra.

        Args:
            asset_id (str): The ID of the asset whose tags will be created.

        Returns:
            Optional[List[str]]: A list of tags if successful, None otherwise.
        """
        endpoint = f"rest/2.0/attributes"
        payload = {"assetId": asset_id, "value": tag}

        try:
            data = self.post(endpoint, json_data=payload, params=None)
            logger.info(f"Successfully created tag with value: {tag} for asset {asset_id}")

    def refresh_database_connections(self, edge_connection_id: UUID, wait_for_completion: bool=True) -> bool:
        """
        Refresh database connections from the data source.
        Retrieve the database connections linked to a specific Edge connection and create the new database connections in Catalog.
        This is an asynchronous API since it needs to reach out to the data source via Edge to retrieve the list of available database connections, which can take some time.

        Args:
            edge_connection_id (UUID): The Edge connection ID to refresh.
            wait_for_completion (bool): Whether to wait for the refresh operation to complete.

        Returns:
            bool: True if the operation is successful, False otherwise.
        """
        if not edge_connection_id or not is_valid_uuid(edge_connection_id):
            logger.error("Invalid edge_connection_id provided. Must be a valid UUID.")
            return False
    
        endpoint = "rest/catalogDatabase/v1/databaseConnections/refresh"
        params = {"edgeConnectionId": edge_connection_id}

        try:
            logger.info(f"Refreshing database connections with edge connection id {edge_connection_id}")
            logger.info(f"Calling endpoint {endpoint} with params: {params}")
            response_data = self.post(endpoint, json_data=None, params=params)
            logger.info(f"Refresh Response: {json.dumps(response_data, indent=4)}")
            
            if not isinstance(response_data, dict) or "id" not in response_data:
                logger.warning("Refreshing database connections initiated, but no valid Job ID was returned.")
                return False
            
            # Extract Job ID
            job_id = response_data.get("id")
            logger.info(f"Refreshing database connections initiated. Job ID: {job_id}")

            if not wait_for_completion:
                return True
            
            # MONITOR JOB STATUS (Async Polling)
            job_status_endpoint = f"rest/2.0/jobs/{job_id}"
            max_retries = 40  # Example: 40 retries with 15-second intervals = 10 minutes
            retry_count = 0

            # Add an initial delay before polling
            time.sleep(self.config.rest_api_sleep)

            while retry_count < max_retries:
                try:
                    # Fetch the job status
                    job_data = self.get(job_status_endpoint)
                    status = job_data.get("state")
                    logger.info(f"Job {job_id} for edge connection {edge_connection_id} current state: {status}")

                    # Check job status
                    if status == "COMPLETED":
                        logger.info("Refreshing database connections successfully completed.")
                        return True
                    elif status in ["CANCELED", "ERROR", "FAILED"]:
                        logger.error(
                            f"Refreshing database connections finished with unsuccessful state: {status}"
                        )
                        return False

                     # Increment retry count and wait before the next poll
                    retry_count += 1
                    time.sleep(self.config.retry_interval_seconds)

                except requests.exceptions.RequestException as e:
                    logger.error(f"Failed to fetch job status for Job ID {job_id}: {e}")
                    return False
                
            # If the job did not complete within the maximum retry limit
            logger.error(f"Job {job_id} did not complete within the maximum retry limit.")
            return False
                
        # If the job did not complete within the maximum retry limit
        except requests.exceptions.RequestException as e:
            logger.error(f"Failed to refresh database connections: {e}")
            return False
        except Exception as e:
            logger.error(f"Unexpected error while refreshing database connections: {str(e)}")
            return False

    def check_asset_by_full_name(self, asset_full_name: str, domain_id: UUID, wait_for_completion: bool = True) -> Optional[Asset]:
        """
        Check if an asset exists by its full name. Optionally wait for its creation.

        Args:
            asset_full_name (str): The full name of the asset to check.
            wait_for_completion (bool): Whether to wait for the asset to be created if it doesn't exist.

        Returns:
            Optional[Asset]: The asset if found, otherwise None.
        """
        params = {
            "name": asset_full_name,
            "nameMatchMode": "END",
            "domainId": domain_id,
            "limit": 1,  # We only need the first match
            "offset": 0,
            "typeInheritance": "true",
            "excludeMeta": "true",
        }

        max_retries = self.config.max_retries # Maximum number of retries (e.g., 80 retries with 15-second intervals = 20 minutes)
        retry_count = 0

        try:
            while True:
                # Make the API call to check for the asset
                res = self.get("rest/2.0/assets", params=params)
                logger.info(f"Response for asset '{asset_full_name}': {json.dumps(res, indent=4)}")
                results = res.get("results", [])

                # If the asset is found, return it
                if results:
                    asset = results[0]
                    try:
                        # Validate that the asset ID is a valid UUID
                        uuid.UUID(asset["id"])
                        logger.info(f"Asset found: {json.dumps(asset, indent=4)}")
                        return Asset(
                            id=asset["id"],
                            full_name=asset["name"],
                            display_name=asset["displayName"],
                            domain_id=asset["domain"]["id"],
                            domain_name=asset["domain"]["name"],
                            type_id=asset["type"]["id"],
                            type_name=asset["type"]["name"],
                            status_id=asset["status"]["id"],
                            status_name=asset["status"]["name"],
                        )
                    except ValueError:
                        logger.error(f"Invalid UUID for asset ID: {asset['id']}")
                        return None

                # If not waiting for completion, return None
                if not wait_for_completion:
                    logger.warning(f"Asset '{asset_full_name}' not found and wait_for_completion is False.")
                    return None

                # Increment retry count and check if the maximum retries have been reached
                retry_count += 1
                if retry_count >= max_retries:
                    logger.error(f"Asset '{asset_full_name}' not found after {max_retries} retries.")
                    return None

                # Wait before the next retry
                logger.info(f"Asset '{asset_full_name}' not found. Retrying in {self.config.retry_interval_seconds} seconds...")
                time.sleep(self.config.retry_interval_seconds)

        except CollibraException as e:
            logger.error(f"Failed to check asset '{asset_full_name}' with error: {e}")
            return None
        except Exception as e:
            logger.error(f"Unexpected error while checking asset '{asset_full_name}': {e}")
            return None

    def trigger_collibra_sync(
        self, 
        database_id: UUID, 
        schema_connection_id: UUID, 
        isStarburst: bool, 
        wait_for_completion: bool = True,
        catalog_schema_path: Optional[str] = None, 
        domain_id: Optional[UUID] = None, ) -> bool:
        """
        Triggers the 'Synchronize Now' command for Collibra Edge and optionally polls for completion.

        Args:
            database_id (UUID): The database ID.
            schema_connection_id (UUID): The schema connection ID.
            catalog_schema_path (str): The catalog schema path.
            domain_id (UUID): The domain ID.
            isStarburst (bool): Whether the source is Starburst.
            wait_for_completion (bool): Whether to wait for the sync to complete.

        Returns:
            bool: True if the sync is successful, False otherwise.
        """
        # 1. TRIGGER SYNC (v1 API)
        sync_endpoint = f"rest/catalogDatabase/v1/databases/{database_id}/synchronizeMetadata"

        payload = (
            # Starburst
            {"schemaConnectionIds": [schema_connection_id]}
            if isStarburst
            # Databricks
            else {
                "domainIncludeMappings": [
                    {
                        "path": catalog_schema_path, 
                        "domainId": domain_id
                        }
                    ],
                "schemaConnectionIds": [schema_connection_id]
                }
            )

        try:
            response_data = self.post(sync_endpoint, json_data=payload, params=None)

            # Extract Job ID
            job_id = response_data.get("id")
            if not job_id:
                logger.warning("Sync initiated, but no Job ID was returned.")
                return True

            logger.info(f"Sync initiated. Job ID: {job_id}")

            if not wait_for_completion:
                return True

            # 2. MONITOR JOB STATUS (Async Polling)
            # Standard 2026 Job API endpoint
            job_status_endpoint = f"rest/2.0/jobs/{job_id}"

            while True:
                job_data = self.get(job_status_endpoint)
                status = job_data.get("state")

                logger.info(f"Collibra Job {job_id} current state: {status}")

                if status == "COMPLETED":
                    logger.info("Collibra Edge synchronization successfully completed.")
                    return True
                elif status in ["CANCELED", "ERROR", "FAILED"]:
                    logger.error(
                        f"Collibra Sync finished with unsuccessful state: {status}"
                    )
                    return False
                
                time.sleep(self.config.retry_interval_seconds)

        except CollibraException as e:
            logger.error(f"Failed to synchronize metadata with database_id: {database_id} and schema_connection_id: {schema_connection_id} {e}")
            return False

    def upsert_classification(self, domain_id: str, full_asset_path: str, classification: str, type_id: UUID) -> Dict[str, Any]:
        """
        Upsert a classification attribute for an asset identified by its full path.

        This function:
        1. Finds the asset by its full path
        2. Gets the asset type ID
        3. Finds the classification attribute type ID
        4. Updates the classification attribute

        Args:
            full_asset_path: The full path of the asset (e.g., 'Domain > Community > Asset Name')
            classification: The classification value to set

        Returns:
            Dict containing the response from the API with attribute details

        Raises:
            CollibraException: If any API operation fails
        """
        logger.info(f"Starting classification upsert for: {full_asset_path}")

        classification_key = str(classification).split(':')[0]
        logger.info(f"classification key: {classification_key}")
        classification_value = str(classification).split(':')[1]
        logger.info(f"classification value: {classification_value}")

        # Step 1: Find the asset by full path
        logger.info(f"Step 1: Finding asset by full path: {full_asset_path}")
        asset = self.get_asset_by_full_path(full_asset_path, domain_id)
        logger.info(f"Asset Info: {asset}")
        asset_id = asset.id
        logger.info(f"asset id: {asset_id}")
        asset_type_id = asset.type_id
        logger.info(f"asset type id: {asset_type_id}")
        
        # Step 2: Find the classification attribute type
        logger.info(f"Step 2: Finding classification attribute type for asset_type_id: {asset_type_id}")
        logger.info(f"classification key: {classification_key}")
        attribute_type_id = self.find_classification_attribute_type_id(classification_key)
        logger.info(f"attribute_type_id: {attribute_type_id}")

        # Step 3: Update classification value for a specific attribute type id under given asset id
        # attribute_type_id 0195065e-4ed8-74c7-b0b4-9f7c5b2bd701 ---> CSI Indicator
        # attribute_type_id:00000000-0000-0000-0001-000500000031 ---> Information Classification

        logger.info(f"Step 3: Updating attribute_type_id: {attribute_type_id} with classification value: {classification_value} under asset id: {asset_id}")

        payload = {
            "typeId": attribute_type_id,
            "values": [classification_value]
        }

        try:
            data = self.put(f"rest/2.0/assets/{asset_id}/attributes", json_data=payload)
            logger.info(f"Successfully updated classification to: {classification}")
            return data
        except CollibraException as e:
            logger.error(f"Failed to update attribute_type_id: {attribute_type_id} with classification value: {classification_value} under asset id: {asset_id} with error {e}")
            raise CollibraException(f"Failed to create attribute: {str(e)}")

        logger.info(f"Classification update is complete.")

class ImmutaException(Exception):

class ImmutaClient(HttpClient):
    def __init__(self, config: ImmutaConfig):
        """
        Initialize the ImmutaClient with the given configuration.

        Args:
            config (ImmutaConfig): Configuration for Immuta API.
        """
        super().__init__(config)  # Call the parent class's constructor

    def _get_headers(self) -> Dict[str, str]:
        """
        Generate headers for Immuta API requests.

        Returns:
            Dict[str, str]: Headers including API key and content type.
        """
        return {
            "Authorization": f"Bearer {self.config.api_key}",
            "Content-Type": "application/json",
            "Accept": "application/json",
        }

    def list_tags(self, search_text: str = None) -> Dict[str, Dict]:
        """
        List all tags in Immuta.
        
        GET /tag
        
        Query Parameters:
        - searchText: Filter returned tags (wildcard prefix/suffix)
        - source: Filter by source
        - includeAllSystemTags: Include deleted system tags
        - limit: Max results
        
        Response: Array of tag objects with id, name, source, deleted, systemCreated, etc.
        
        Returns:
            Dict mapping tag names (lowercase) to tag details
        """
        try:
            params = {}
            if search_text:
                params['searchText'] = search_text
            
            response = self.get("tag", params=params if params else None)
            
            # Response is an array of tag objects
            tags_list = response
            
            # Handle if wrapped in an object
            if isinstance(tags_list, dict):
                tags_list = tags_list.get('tags', tags_list.get('data', []))
            
            # Filter out deleted tags and map tag names to their details
            result = {tag['name'].lower(): tag for tag in tags_list if not tag.get('deleted', False)}
            
            logger.info(f"Retrieved {len(result)} tags from Immuta")
            return result
            
        except ImmutaException as e:
            logger.error(f"Failed to list Immuta tags: {e}")

    def tag_exists(self, tag_name: str) -> bool:
        """Check if a tag exists in Immuta."""
        tags = self.list_tags()
        return tag_name.lower() in tags

    def trigger_immuta_sync_v_2(self, host: str, connection_key: str, catalog_name: str, schema_name: str) -> Dict[str, Any]:

            )

                return {"success": False, "message": "No bulk ID returned"}

    def get_data_source_id(self, catalog: str, schema: str, table: str, blob_handler_type: str) -> Optional[int]:
        # Databricks: "blobHandlerType": "Databricks"
        # Starburst: "blobHandlerType": "Trino"
        params = {

            "searchText": table,
            "nameOnly": "true",
            "blobHandlerType": blob_handler_type, 
        }
        # Retrieve everything for testing
        # params = {}
        try:
            data = self.get("dataSource", params=params)
            logger.info(f"Immuta data source: {data}")
            hits = data.get("hits", [])
            for ds in hits:
                if blob_handler_type == "Databricks":
                    if (ds.get("remoteTable") == table and ds.get("remoteSchema") == schema and catalog in ds.get("name", "")):

                    if (ds.get("remoteTable") == table and ds.get("remoteSchema") == schema):
                        return ds["id"]
                
            return None
        except ImmutaException as e:
            logger.error(f"Failed to fetch data source ID: {e}")
            raise
    
    def get_bulk_operation_status(self, bulk_id: str) -> Dict[str, Any]:
        """
        Fetch and monitor the status of a bulk operation in Immuta until it completes.

        Args:
            bulk_id (str): The ID of the bulk operation.

        Returns:
            Dict[str, Any]: The inferred status of the bulk operation, including:
                - success (bool): Whether the operation was successful.
                - status (str): The inferred status (e.g., "COMPLETED", "FAILED").
                - details (Dict): The raw response from the API.
        """
        try:
            params = {"bulkId": bulk_id}

            while True:
                logger.info(f"Fetching status for bulk operation ID: {bulk_id}")
                response = self.get(f"jobs", params=params)
                logger.info(f"Bulk operation status: {json.dumps(response, indent=4)}")

                # Extract relevant fields from the response
                total = response.get("total", 0)
                completed = response.get("completed", 0)
                failed = response.get("failed", 0)
                pending = response.get("pending", 0)
                errors = response.get("errors")

                # Check if the operation is complete
                if pending == 0:
                    if total == completed and failed == 0 and errors is None:
                        logger.info(f"Bulk operation {bulk_id} completed successfully.")
                        return {
                            "success": True,
                            "status": "COMPLETED",
                            "details": response,
                        }
                    else:
                        logger.error(f"Bulk operation {bulk_id} failed or incomplete.")
                        return {
                            "success": False,
                            "status": "FAILED",
                            "details": response,
                        }

                # Wait for a short interval before checking again
                time.sleep(immuta_config.rest_api_sleep)

        except ImmutaException as e:
            logger.error(f"Failed to fetch bulk operation status for ID {bulk_id}: {e}")
            return response

    def create_tag(self, tag_name: str, description: Optional[str] = None) -> Dict:
        """
        Create a new tag in Immuta.

        POST /tag

        Payload:
        {
            "tags": [{"name": "Parent.Child"}]
        }

        For hierarchical tags, use dot notation: "Parent.Child.Grandchild"

        Args:
            tag_name (str): Name of the tag to create (can include dots for hierarchy).
            description (Optional[str]): Optional description for the tag.

        Returns:
            Dict: Details of the created tag, or an empty dictionary if the operation fails.
        """
        if "." in tag_name:
            # For hierarchical tags, define the root (e.g., 'CDP') 
            # to force Immuta to index the parent-child relationship
            root_name = tag_name.split('.')[0]
            payload = {
                "rootTag": {"name": root_name, "deleteHierarchy": False},
                "tags": [{"name": tag_name, "source": "curated"}]
            }
        else:
            # For flat tags, a simple list works best
            payload = {"tags": [{"name": tag_name, "source": "curated"}]}

        try:
            # Send the POST request to create the tag
            response = self.post("tag", json_data=payload)

            # Log success and parse the response
            logger.info(f"Successfully created tag '{tag_name}' in Immuta.")
            result = response

            # Return the first tag if the response is a list, otherwise return the response
            return result[0] if isinstance(result, list) and result else result

        except ImmutaException as e:
            # Log the error and return an empty dictionary
            logger.error(f"Failed to create tag '{tag_name}' in Immuta: {e}")
            return {}

        except Exception as e:
            # Handle unexpected exceptions
            logger.error(f"Unexpected error while creating tag '{tag_name}': {e}")
            return {}

def _try_call_patterns(sync_fn, domain: str, catalog: str, schema: str, obj: str,
                       tags: List[str], classifications: Dict[str, Any], is_starburst: bool):
    """
    Try multiple call signatures to tolerate minor API differences:
    - isStarburst=...
    - is_starburst=...
    - with or without tags/classifications
    """
    patterns = [
        ("domain,catalog,schema,obj,tags,cls,isStarburst=..",
         lambda: sync_fn(domain, catalog, schema, obj, tags, classifications, isStarburst=is_starburst)),
        ("domain,catalog,schema,obj,tags,cls,is_starburst=..",
         lambda: sync_fn(domain, catalog, schema, obj, tags, classifications, is_starburst=is_starburst)),
        ("domain,catalog,schema,obj,tags,cls",
         lambda: sync_fn(domain, catalog, schema, obj, tags, classifications)),
        ("domain,catalog,schema,obj,isStarburst=..",
         lambda: sync_fn(domain, catalog, schema, obj, isStarburst=is_starburst)),
        ("domain,catalog,schema,obj,is_starburst=..",
         lambda: sync_fn(domain, catalog, schema, obj, is_starburst=is_starburst)),
        ("domain,catalog,schema,obj",
         lambda: sync_fn(domain, catalog, schema, obj)),
    ]

    last_err = None
    for label, fn in patterns:
        try:
            result = fn()
            print(f"✅ sync_platform.sync succeeded using pattern: {label}")
            return result
        except TypeError as e:
            last_err = e
        except Exception:
            raise

    raise TypeError(f"Unable to call sync_platform.sync with any known signature. Last TypeError: {last_err}")

def _sql_literal(s: str) -> str:
    return "'" + s.replace("'", "''") + "'"


def _quote_ident(part: str) -> str:
    part = part.replace("`", "``")
    return f"`{part}`"


def _normalize_fqn(catalog: str, schema: str, object_name: str) -> Tuple[str, str, str]:
    """
    Accepts:
      - object
      - schema.object
      - catalog.schema.object
    """
    parts = [p for p in object_name.split(".") if p]
    if len(parts) == 1:
        if not catalog or not schema:
            raise ValueError("If object_name is not fully qualified, catalog and schema must be provided.")
        return catalog, schema, parts[0]
    if len(parts) == 2:
        if not catalog:
            raise ValueError("If object_name is schema.object, catalog must be provided.")
        return catalog, parts[0], parts[1]
    if len(parts) == 3:
        return parts[0], parts[1], parts[2]
    raise ValueError("object_name must be in form object, schema.object, or catalog.schema.object")


def _detect_object_type(catalog: str, schema: str, obj: str) -> str:
    df = spark.sql(f"""
        SELECT table_type
        FROM system.information_schema.tables
        WHERE table_catalog = {_sql_literal(catalog)}
          AND table_schema  = {_sql_literal(schema)}
          AND table_name    = {_sql_literal(obj)}
        LIMIT 1
    """)
    rows = df.collect()
    if not rows:
        raise ValueError(f"Object not found in system.information_schema.tables: {catalog}.{schema}.{obj}")
    table_type = (rows[0]["table_type"] or "").upper()
    if "VIEW" in table_type:
        return "VIEW"
    return "TABLE"


def _get_current_owner(catalog: str, schema: str, obj: str):
    df = spark.sql(f"""
        SELECT table_owner
        FROM system.information_schema.tables
        WHERE table_catalog = {_sql_literal(catalog)}
          AND table_schema  = {_sql_literal(schema)}
          AND table_name    = {_sql_literal(obj)}
        LIMIT 1
    """)
    rows = df.collect()
    if not rows:
        return None
    return rows[0]["table_owner"]

def _as_bool(s: str) -> bool:
    return str(s).strip().lower() in {"1", "true", "t", "yes", "y"}

def _parse_tags(tags_csv: str) -> Set[str]:
    if not tags_csv or not tags_csv.strip():
        return set()
    # split by comma OR newline
    raw = re.split(r"[\n,]+", tags_csv)
    return {t.strip() for t in raw if t and t.strip()}

def _parse_classifications(classifications_json: str) -> Dict[str, Any]:
    if not classifications_json or not classifications_json.strip():
        return {}
    try:
        parsed = json.loads(classifications_json)
    except Exception as e:
        raise ValueError(f"Got error: {e}")
    if not isinstance(parsed, dict):
        raise ValueError("classifications_json must decode to a JSON object/dict.")
    return parsed

import types

def _try_call_patterns(sync_fn, domain: str, catalog: str, schema: str, obj: str,
                    tags: Set[str], classifications: Dict[str, Any], is_starburst: bool):
    """Try multiple call signatures to be robust against minor API differences."""
    # to study what the API sends to the user
    patterns = [
        ("domain,catalog,schema,obj,tags,cls,isStarburst=..",
        lambda: sync_fn(domain, catalog, schema, obj, tags, classifications, isStarburst=is_starburst)),
        ("domain,catalog,schema,obj,tags,cls,is_starburst=..",
        lambda: sync_fn(domain, catalog, schema, obj, tags, classifications, is_starburst=is_starburst)),
        ("domain,catalog,schema,obj,tags,cls",
        lambda: sync_fn(domain, catalog, schema, obj, tags, classifications)),
        ("domain,catalog,schema,obj,isStarburst=..",
        lambda: sync_fn(domain, catalog, schema, obj, isStarburst=is_starburst)),
        ("domain,catalog,schema,obj,is_starburst=..",
        lambda: sync_fn(domain, catalog, schema, obj, is_starburst=is_starburst)),
        ("domain,catalog,schema,obj",
        lambda: sync_fn(domain, catalog, schema, obj)),
    ]

    last_err = None
    for label, fn in patterns:
        try:
            res = fn()
            print(f"✅ sync_platform.sync succeeded using pattern: {label}")
            return res
        except TypeError as e:
            last_err = e
            # keep trying
        except Exception:
            # non-TypeError likely means the call signature was accepted but the function failed internally;
            # re-raise to surface the true issue.
            raise

    raise TypeError(f"Unable to call sync_platform.sync with any known signature. Last TypeError: {last_err}")
