Deployment Instructions:

# 1. Clean up any existing instances
pkill -f "python.*overlay.py" 2>/dev/null

# 2. Setup environment
python -m venv venv
source venv/bin/activate
pip install -r requirements.txt

# 3. First test run (debug mode)
python overlay.py

# 4. Production deployment
nohup python overlay.py > overlay.log 2>&1 &
disown

# 5. Verification
pgrep -fl "python.*overlay.py"  # More precise process check
tail -f overlay.log  # View logs in real-time (Ctrl+C to exit)
